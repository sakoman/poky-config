update any needed last-revision in .conf file
./buildit-nut nanbield
cd ../poky
git push -f contrib stable/nanbield-nut

Log into builders:

$ ssh sakoman@debian8-ty-1.yocto.io
$ sudo -iu pokybuild

sstate-cache-management.sh --cache-dir=/home/steve/yocto-files/nanbield/sstate -v -d

/usr/bin/find /home/steve/yocto-files/nanbield/sstate-cache \( -mtime +21 -a -atime +41 \) -type f \( -name '*.tgz' -o -name '*tgz.siginfo' \) -delete
